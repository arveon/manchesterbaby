;will add 50 to 200, multiply result by 2 and store in Res
VAR 0;
        LDN 200;//stores -200
        SUB 50;//adds -50 (-250)
        MUL Multiplier;//multiplies by 2 (-500)
        DIV 5; //divides by 5 (-100)
        STO Res;
;        
        LDN Res; //negates result (100)
        STO Res; //stores in Res (100)
        STP;
;
Res: VAR 0;
Multiplier: VAR 2;
